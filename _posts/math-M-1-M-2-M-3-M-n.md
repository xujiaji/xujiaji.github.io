$$
\begin{align}
& 　　　　　　令： S = M^1 + M^2 + M^3 + ... + M^n \\
& 两边乘以Ｍ，则： S \times M = M^2 + M^3 + ... + M^n + M^{n+1} \\
& 　　　　　　则： S \times M = -M^1 + M^1 + M^2 + M^3 + ... + M^n + M^{n+1} \\
& 　　　　因此有： S \times M = -M^1 + S + M^{n+1} \\ 
& 　　　　　则有： S \times (M - 1) = M^{n+1} - M 　可得 \\
& 　　　求和公式： S = {{M \times (M^n -1)} \over {M - 1}}
\end{align}
$$
